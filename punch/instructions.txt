    - Install node and git globally
    - Run node command prompt as admin
    - npm install -g punch
    - Sparse checkout:
        git init aerocoat
        cd aerocoat
        git remote add -f origin https://brophdawg11@bitbucket.org/brophdawg11/aerocoat.git
        git config core.sparsecheckout true
        echo punch/*> .git/info/sparse-checkout
          (Note: /* is required, as is no space before >)
        git pull origin master
        git branch -u origin/master
        cd punch
        npm install
        -- copy "All Suppliers -TB and MSDS" directory to templates/ --
        npm run clean
        npm run csv
        npm run generate
        npm run serve-output
          -- Make sure site looks OK --

        -- Install aws cli for windows --
        -- Close and reopen command prompt to get new PATH --
        aws configure --profile aerocoat
        npm run sync-test
